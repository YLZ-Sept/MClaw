## 二、后端完整代码（Node.js + Express）

### 直接复制，可直接运行

```javascript
// server.js
const express = require('express');
const cors = require('cors');
const app = express();
app.use(cors());
app.use(express.json());
// 模拟聊天记录
let chatHistory = [
 { role: 'user', content: 'E:\CC\蛙蔻面板' },
 { role: 'ai', content: '已记住路径，下次生成文件直接保存到E:\CC\蛙蔻面板' }
];
// 1. 获取系统信息
app.get('/api/info', (req, res) => {
 res.json({
 code: 200,
 data: {
 version: 'v0.14.0',
 engine: 'OpenClaw',
 status: 'running'
 }
 });
});
// 2. 获取聊天历史
app.get('/api/chat/history', (req, res) => {
 res.json({ code: 200, data: chatHistory });
});
// 3. 发送消息
app.post('/api/chat/send', (req, res) => {
 const { content } = req.body;
 chatHistory.push({ role: 'user', content });

// AI 自动回复
 let reply = '';
 if (content.includes('Desktop') || content.includes('桌面')) {
 reply = '已记住桌面路径，以后生成 Excel/Word/PPT 自动保存到桌面';
 } else {
 reply = `已收到：${content}，OpenClaw 正在处理...`;
 }
 chatHistory.push({ role: 'ai', content: reply });
 res.json({ code: 200, data: { content: reply } });
});
// 4. 系统状态
app.get('/api/status', (req, res) => {
 res.json({
 code: 200,
 data: { cpu: '12%', memory: '380MB', services: 3 }
 });
});
const PORT = 3666;
app.listen(PORT, () => {
 console.log(`蛙蔻面板后端运行在 http://localhost:${PORT}`);
});
```

### 启动命令

```bash
npm init -y
npm install express cors
node server.js
```

## 三、前端 API 对接文件（直接用）

```javascript
// src/api/index.js
import axios from 'axios'
const request = axios.create({
 baseURL: 'http://localhost:3666/api'
})
// 获取系统信息
export function getInfo() {
 return request.get('/info')
}
// 获取聊天记录
export function getChatHistory() {
 return request.get('/chat/history')
}
// 发送消息
export function sendMessage(content) {
 return request.post('/chat/send', { content })
}
// 获取系统状态
export function getStatus() {
 return request.get('/status')
}
```
